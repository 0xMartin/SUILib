SUILib.application

Main application and view management for SUILib

This module provides the core Application and View classes for the SUILib framework, supporting multiple switchable views, style management, event loop, window handling, and scheduling of repaint and rendering logic.

Classes:

Application: Main class managing the application window, views, rendering, and events. View: Abstract base class for a content page shown in the application window. Layout: Abstract base class for layout managers that arrange GUI elements in a view.

185    def run(self, start_view=None) -> bool:
186        """
187        Start the main event loop and rendering.
188
189        Args:
190            start_view (View, optional): View to show first.
191
192        Returns:
193            bool: True if the loop exited normally, False if not initialized.
194        """
195        if not self.inited:
196            return False
197
198        self.running = True
199
200        # call start event for each view
201        for view in self.views:
202            view.createEvt_base(self.screen.get_width(), self.screen.get_height())
203
204        if start_view is not None:
205            self.showView(start_view)
206
207        # Enable periodic repaint if requested
208        if self._periodic_repaint_enabled:
209            pygame.time.set_timer(REPAINT_EVENT, int(1000 / self._periodic_repaint_fps))
210
211        while self.running:
212            event = pygame.event.wait()
213
214            if event.type == pygame.QUIT:
215                self.running = False
216            elif event.type == REPAINT_EVENT:
217                self._needs_repaint = True
218            else:
219                if self.visible_view is not None:
220                    self.visible_view.processEvt(event)
221                    self.visible_view.update()
222                self._needs_repaint = True  # repaint after every event
223
224            if self._needs_repaint and self.visible_view is not None:
225                if self.visible_view.getFillColor() is None:
226                    self.screen.fill(self.fill_color)
227                else:
228                    self.screen.fill(self.visible_view.getFillColor())
229                self.visible_view.render(self.screen)
230                pygame.display.flip()
231                self._needs_repaint = False  # repaint done
232
233        pygame.quit()
234        return True

236    def requestRepaint(self):
237        """
238        Request an immediate repaint of the active view.
239        """
240        self._needs_repaint = True
241        if self.visible_view is not None:
242            if self.visible_view.getFillColor() is None:
243                self.screen.fill(self.fill_color)
244            else:
245                self.screen.fill(self.visible_view.getFillColor())
246            self.visible_view.render(self.screen)
247            pygame.display.flip()
248            self._needs_repaint = False

280    def showView(self, view) -> bool:
281        """
282        Display a specific view in the application.
283
284        Args:
285            view (View): The view to display.
286
287        Returns:
288            bool: True on success, False otherwise.
289        """
290        if not self.running:
291            return False
292
293        if view in self.views:
294            # hide current visible view
295            if self.visible_view is not None:
296                self.visible_view.setVisibility(False)
297                self.visible_view.hideEvt()
298            # show new view
299            view.setVisibility(True)
300            view.openEvt_base(self.screen.get_width(),
301                              self.screen.get_height())
302            self.visible_view = view
303            # change window title
304            if len(view.name) == 0:
305                pygame.display.set_caption(self.name)
306            else:
307                pygame.display.set_caption(self.name + " - " + view.name)
308            return True
309        else:
310            return False

355    """
356    Abstract base class representing a single page or content view in the application window.
357
358    The View class defines the interface and core logic for all views in the SUILib framework. 
359    A view is a logical page or screen which the user can see and interact with; only one view 
360    is visible at a time. Each view manages its own GUI elements, layouts, appearance, and event 
361    handling, and can be shown, hidden, or updated by the Application.

635    @final
636    def processEvt(self, event):
637        """
638        Process a single event from the application and dispatch to GUI elements.
639
640        Args:
641            event: A pygame event object.
642        """
643        if self.app is not None:
644            if self.filter is None:
645                for el in self.GUIElements:
646                    el.processEvent(self, event)
647            else:
648                self.filter["element"].processEvent(self, event)
649        selected = self.findElement(
650            self.GUIElements, lambda el: el.isSelected())
651        if selected is not None:
652            pygame.mouse.set_cursor(selected.getSelectCursor())
653        else:
654            pygame.mouse.set_cursor(self.cursor)

656    def findElement(self, list, procces_function=None):
657        """
658        Find the first element in a list of GUI elements for which a process function returns True.
659
660        Args:
661            list (list): List of GUIElement (or Container) objects.
662            procces_function (callable): Function accepting a GUIElement and returning bool.
663
664        Returns:
665            GUIElement or None: The first matching element, or None if not found.
666        """
667        if procces_function is None or list is None:
668            return None
669        ret = None
670        for el in list:
671            if isinstance(el, Container):
672                ret_container = self.findElement(
673                    el.getChilds(), procces_function)
674                if ret_container is not None:
675                    ret = ret_container
676                    break
677            else:
678                if procces_function(el):
679                    ret = el
680                    break
681        return ret

708class Layout(metaclass=abc.ABCMeta):
709    """
710    Abstract base class for layout managers in SUILib.
711
712    The Layout class manages a list of layout elements, each consisting of a reference to a GUI element 
713    and its properties relevant to the specific layout manager. 
714    Layout elements are stored as dictionaries: {"element": ..., "propt": ...}.
715
716    Subclasses should implement the updateLayout() method to arrange elements according to their own rules.
717
718    Attributes:
719        view (View): The associated View instance for this layout manager.
720        layoutElements (list): List of layout elements managed by this layout manager.
721    """
722
723    def __init__(self, view: View, register: bool = True):
724        """
725        Initialize the base layout class and optionally register it with the view.
726
727        Args:
728            view (View): The View instance this layout manager is associated with.
729            register (bool): If True (default), automatically register with the view.
730        """
731        if isinstance(view, View):
732            self.view = view
733        self.layoutElements = []
734        # register
735        if register:
736            view.registerLayoutManager(self)
737
738    @final
739    def getLayoutElements(self) -> list:
740        """
741        Get the list of layout elements managed by this layout manager.
742
743        Returns:
744            list: List of layout element dictionaries.
745        """
746        return self.layoutElements

748    def setElements(self, layoutElements):

Abstract base class representing a single page or content view in the application window.

The View class defines the interface and core logic for all views in the SUILib framework. A view is a logical page or screen which the user can see and interact with; only one view is visible at a time. Each view manages its own GUI elements, layouts, appearance, and event handling, and can be shown, hidden, or updated by the Application.

Subclasses should implement the updateLayout() method to arrange elements according to their own rules.
